# Heir of Wrong Division
country_event = {
	id = kor_factionalism.1
	title = "kor_factionalism.1.t"
	desc = "kor_factionalism.1.d"
	picture = IMPERIAL_SEAL_eventPicture

	trigger = {
		has_dlc = "Res Publica"
		has_estate = estate_nobles
		has_reform = joseon_monarchy_2
		has_consort = yes
		is_female = no
		has_consort_regency = no
		has_heir = no
		OR = {
			statists_vs_orangists = 0.5
			NOT = { statists_vs_orangists = -0.5 }
		}
	}

	mean_time_to_happen = {
		months = 360
	}

	# He is my rightful heir
	option = {
		name = "kor_factionalism.1.a"
		
		define_heir = {
			male = yes
			dynasty = ROOT
			age = 0
			claim = 50
			no_consort_with_heir = yes
		}
		if = {
			limit = {
				statists_vs_orangists = 0.5
			}
			set_heir_flag = kor_biased_west
		}
		else = {
			set_heir_flag = kor_biased_east
		}
		set_heir_flag = kor_heir_wrong_division
		add_estate_loyalty = {
			estate = estate_nobles
			loyalty = -15
		}
		hidden_effect = {
			if = {
				limit = {
					NOT = { has_global_flag = Kor_Huibin_Jang }
				}
				country_event = { 
					id = kor_factionalism.7
					days = 730
					random = 30
				}
			}
		}
		ai_chance = {
			factor = 1
			modifier = {
				factor = 0
				NOT = { estate_loyalty = { estate = estate_noble loyalty = 40 } }
			}
		}
	}

	# Wonderful, but he can't be my rightful heir.
	option = {
		name = "kor_factionalism.1.b"
		if = {
			limit = {
				statists_vs_orangists = 0.5
			}
			change_statists_vs_orangists = 0.15
		}
		else = {
		change_statists_vs_orangists = -0.15
		}
		ai_chance = {
			factor = 1
		}
	}
}

country_event = {
	id = kor_factionalism.4
	title = "kor_factionalism.4.t"
	desc = "kor_factionalism.4.d"
	picture = MILITARY_CAMP_eventPicture

	trigger = {
		always = no
		NOT = { has_country_modifier = war_against_joseon }
		OR = {
			tag = MCH
			tag = JAP
		}
		KOR = {
			AND = {
				exists = yes
				has_reform = joseon_monarchy_2
				NOT = { is_subject_of = ROOT }
			}
		}
		NOT = { alliance_with = KOR }
		NOT = { marriage_with = KOR }
		NOT = { is_emperor_of_china = yes }
	}

	mean_time_to_happen = {
		months = 360
	}

	# Prepare for war.
	option = {
		name = "kor_factionalism.4.a"
		add_treasury = -150
		add_country_modifier = {
			name = war_against_joseon
			duration = 3650
		}
		hidden_effect = {
			if = {
				limit = {
					tag = MCH
				}	
			KOR = { country_event = { id = kor_factionalism.4 days = 183 } }
			}
			else = {

			KOR = { country_event = { id = kor_factionalism.3 days = 183 } }

			}
		}
		ai_chance = {
			factor = 1
			modifier = {
				factor = 0
				NOT = { treasury = 100 }
			}
			modifier = {
				factor = 0
				NOT = {
					army_strength = {
						who = KOR
						value = 0.66
					}
				}
			}
			modifier = {
				factor = 0.1
				OR = {
					ai_attitude = {
						who = KOR
						attitude = attitude_friendly
					}
					ai_attitude = {
						who = KOR
						attitude = attitude_threatened
					}
				}
			}
			modifier = {
				factor = 100
				OR = {
					is_rival = KOR
					ai_attitude = {
						who = KOR
						attitude = attitude_outraged
					}
					ai_attitude = {
						who = KOR
						attitude = attitude_hostile
					}
					ai_attitude = {
						who = KOR
						attitude = attitude_domineering
					}
				}
			}
		}
	}

	# Not right now.
	option = {
		name = "kor_factionalism.4.b"
		ai_chance = {
			factor = 1
		}
	}
}
